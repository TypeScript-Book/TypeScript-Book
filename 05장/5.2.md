# 5.2

## 5.2.1

```typescript
interface Clock {
  time: Date;
  setTime(date: Date): void;
}

function setClockTime(this: Clock, date: Date): void {
  this.time = date;
}

const clock: Clock = {
  time: new Date(),
  setTime: setClockTime,
};

clock.setTime(new Date());

console.log(clock.time);
```

## 5.2.2

```typescript
interface Clock {
  time: Date;
  setTime(date: Date): void;
  getCurrentTime(): Date;
}

const clock: Clock = {
  time: new Date(),
  setTime(date: Date) {
    this.time = date;
  },
  getTime() {
    return this.time;
  },
};

clock.setTime(new Date());
console.log(clock.getTime());
```
